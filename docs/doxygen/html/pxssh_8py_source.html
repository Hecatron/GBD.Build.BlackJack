37 # This class extends pexpect.spawn to specialize setting up SSH
38 # connections. This adds methods for login, logout, and expecting the shell
39 # prompt. It does various tricky things to handle many situations in the SSH
40 # login process. For example, if the session is your first login, then pxssh
41 # automatically accepts the remote certificate; or if you have public key
42 # authentication setup then pxssh won't wait for the password prompt.
43 #
44 # pxssh uses the shell prompt to synchronize output from the remote host. In
45 # order to make this more robust it sets the shell prompt to something more
46 # unique than just $ or #. This should work on most Borne/Bash or Csh style
47 # shells.
48 #
49 # Example that runs a few commands on a remote server and prints the result::
50 #
51 # import pxssh
52 # import getpass
53 # try:
54 # s = pxssh.pxssh()
55 # hostname = raw_input('hostname: ')
56 # username = raw_input('username: ')
57 # password = getpass.getpass('password: ')
58 # s.login(hostname, username, password)
59 # s.sendline('uptime') # run a command
60 # s.prompt() # match the prompt
61 # print(s.before) # print everything before the prompt.

64 # print(s.before)
65 # s.sendline('df')

67 # print(s.before)
68 # s.logout()
69 # except pxssh.ExceptionPxssh as e:
70 # print("pxssh failed on login.")

74 #

77 # "StrictHostKeyChecking": "no",

81 # Note that if you have ssh-agent running while doing development with pxssh
82 # then this can lead to a lot of confusion. Many X display managers (xdm,
83 # gdm, kdm, etc.) will automatically start a GUI agent. You may see a GUI
84 # dialog box popup asking for a password during development. You should turn

140  # This calculates the Levenshtein distance between a and b.
141  #
142  def levenshtein_distance(self, a, b):
143 
144  n, m = len(a), len(b)
145  if n > m:
146  a,b = b,a
147  n,m = m,n
148  current = range(n+1)
149  for i in range(1,m+1):
150  previous, current = current, [i]+[0]*n
151  for j in range(1,n+1):
152  add, delete = previous[j]+1, current[j-1]+1
153  change = previous[j-1]
154  if a[j-1] != b[i-1]:
155  change = change + 1
156  current[j] = min(add, delete, change)
157  return current[n]

160  # This facilitates using communication timeouts to perform
161  # synchronization as quickly as possible, while supporting high latency
162  # connections with a tunable worst case performance. Fast connections
163  # should be read almost immediately. Worst case performance for this
164  # method is timeout_multiplier * 3 seconds.

166  def try_read_prompt(self, timeout_multiplier):
167 
168  # maximum time allowed to read the first response
169  first_char_timeout = timeout_multiplier * 0.5
170 
171  # maximum time allowed between subsequent characters
172  inter_char_timeout = timeout_multiplier * 0.1
173 
174  # maximum time for reading the entire prompt
175  total_timeout = timeout_multiplier * 3.0
176 
177  prompt = b''
178  begin = time.time()
179  expired = 0.0

181

183  try:
184  prompt += self.read_nonblocking(size=1, timeout=timeout)
185  expired = time.time() - begin # updated total time expired
186  timeout = inter_char_timeout

234  # This logs the user into the given server.
235  #
236  # It uses
237  # 'original_prompt' to try to find the prompt right after login. When it
238  # finds the prompt it immediately tries to reset the prompt to something
239  # more easily matched. The default 'original_prompt' is very optimistic
240  # and is easily fooled. It's more reliable to try to match the original
241  # prompt as exactly as possible to prevent false matches by server
242  # strings such as the "Message Of The Day". On many systems you can
243  # disable the MOTD on the remote server by creating a zero-length file
244  # called :file:`~/.hushlogin` on the remote server. If a prompt cannot be found
245  # then this will not necessarily cause the login to fail. In the case of
246  # a timeout when looking for the prompt we assume that the original
247  # prompt was so weird that we could not match it, so we use a few tricks
248  # to guess when we have reached the prompt. Then we hope for the best and
249  # blindly try to reset the prompt to something more unique. If that fails
250  # then login() raises an :class:`ExceptionPxssh` exception.
251  #
252  # In some situations it is not possible or desirable to reset the
253  # original prompt. In this case, pass ``auto_prompt_reset=False`` to
254  # inhibit setting the prompt to the UNIQUE_PROMPT. Remember that pxssh
255  # uses a unique prompt in the :meth:`prompt` method. If the original prompt is
256  # not reset then this will disable the :meth:`prompt` method unless you
257  # manually set the :attr:`PROMPT` attribute.
258  #

386  ##
387  # This sets the remote prompt to something more unique than ``#`` or ``$``.
388  # This makes it easier for the :meth:`prompt` method to match the shell prompt
389  # unambiguously. This method is called automatically by the :meth:`login`
390  # method, but you may want to call it manually if you somehow reset the
391  # shell prompt. For example, if you 'su' to a different user then you
392  # will need to manually reset the prompt. This sends shell commands to
393  # the remote host to set the prompt, so this assumes the remote host is
394  # ready to receive commands.
395  #
396  # Alternatively, you may use your own prompt pattern. In this case you
397  # should call :meth:`login` with ``auto_prompt_reset=False``; then set the
398  # :attr:`PROMPT` attribute to a regular expression. After that, the
399  # :meth:`prompt` method will try to match your prompt pattern.
400  #
401  def set_unique_prompt(self):
402 
403  self.sendline("unset PROMPT_COMMAND")
404  self.sendline(self.PROMPT_SET_SH) # sh-style
405  i = self.expect ([TIMEOUT, self.PROMPT], timeout=10)
406  if i == 0: # csh-style
407  self.sendline(self.PROMPT_SET_CSH)
408  i = self.expect([TIMEOUT, self.PROMPT], timeout=10)
409  if i == 0:
410  return False
411  return True
