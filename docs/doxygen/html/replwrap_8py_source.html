23 # Wrapper for a REPL.
24 #
25 # :param cmd_or_spawn: This can either be an instance of :class:`pexpect.spawn`
26 # in which a REPL has already been started, or a str command to start a new
27 # REPL process.
28 # :param str orig_prompt: The prompt to expect at first.
29 # :param str prompt_change: A command to change the prompt to something more
30 # unique. If this is ``None``, the prompt will not be changed. This will
31 # be formatted with the new and continuation prompts as positional
32 # parameters, so you can use ``{}`` style formatting to insert them into
33 # the command.
34 # :param str new_prompt: The more unique prompt to expect after the change.
35 # :param str extra_init_cmd: Commands to do extra initialisation, such as
36 # disabling pagers.
37 #
38 class REPLWrapper(object):

40  new_prompt=PEXPECT_PROMPT,

75  # Send a command to the REPL, wait for and return output.
76  #
77  # :param str command: The command to send. Trailing newlines are not needed.
78  # This should be a complete block of input that will trigger execution;
79  # if a continuation prompt is found after sending input, :exc:`ValueError`
80  # will be raised.
81  # :param int timeout: How long to wait for the next prompt. -1 means the
82  # default from the :class:`pexpect.spawn` object (default 30 seconds).
83  # None means to wait indefinitely.
84  #
85  def run_command(self, command, timeout=-1):
86  # Split up multiline commands and feed them in bit-by-bit
87  cmdlines = command.splitlines()
88  # splitlines ignores trailing newlines - add it back in manually
89  if command.endswith('\n'):
90  cmdlines.append('')
91  if not cmdlines:
92  raise ValueError("No command was given")
93 
94  self.child.sendline(cmdlines[0])
95  for line in cmdlines[1:]:
96  self._expect_prompt(timeout=1)
97  self.child.sendline(line)
98 
99  # Command was fully submitted, now wait for the next prompt
100  if self._expect_prompt(timeout=timeout) == 1:
101  # We got the continuation prompt - command was incomplete
102  self.child.kill(signal.SIGINT)
103  self._expect_prompt(timeout=1)
104  raise ValueError("Continuation prompt found - input was incomplete:\n"
105  + command)
106  return self.child.before
