309  # This puts a character at the current cursor position. The cursor
310  # position is moved forward with wrap-around, but no scrolling is done if
311  # the cursor hits the lower-right corner of the screen.
312  def write_ch (self, ch):
313 
314  if isinstance(ch, bytes):
315  ch = self._decode(ch)
316 
317  #\r and \n both produce a call to cr() and lf(), respectively.
318  ch = ch[0]
319 
320  if ch == u'\r':
321  self.cr()
322  return
323  if ch == u'\n':
324  self.crlf()
325  return
326  if ch == chr(screen.BS):
327  self.cursor_back()
328  return
329  self.put_abs(self.cur_r, self.cur_c, ch)
330  old_r = self.cur_r
331  old_c = self.cur_c
332  self.cursor_forward()
333  if old_c == self.cur_c:
334  self.cursor_down()
335  if old_r != self.cur_r:
336  self.cursor_home (self.cur_r, 1)
337  else:
338  self.scroll_up ()
339  self.cursor_home (self.cur_r, 1)
340  self.erase_line()
