78 # Split a command line into a command and its arguments according to
79 # the rules of the Microsoft C runtime.
80 def split_command_line(cmdline):
81  # http://msdn.microsoft.com/en-us/library/ms880421
82  s_free, s_in_quotes, s_in_escape = list(range(3))
83  state = namedtuple('state',
84  ('current', 'previous', 'escape_level', 'argument'))
85  state.current = s_free
86  state.previous = s_free
87  state.argument = []
88  result = []
89  for c in itertools.chain(cmdline, ['EOI']): # Mark End of Input
90  if state.current == s_free:
91  if c == '"':
92  state.current = s_in_quotes
93  state.previous = s_free
94  elif c == '\\':
95  state.current = s_in_escape
96  state.previous = s_free
97  state.escape_count = 1
98  elif c in (' ', '\t', 'EOI'):
99  if state.argument or state.previous != s_free:
100  result.append(''.join(state.argument))
101  del state.argument[:]
102  else:
103  state.argument.append(c)
104  elif state.current == s_in_quotes:
105  if c == '"':
106  state.current = s_free
107  state.previous = s_in_quotes
108  elif c == '\\':
109  state.current = s_in_escape
110  state.previous = s_in_quotes
111  state.escape_count = 1
112  else:
113  state.argument.append(c)
114  elif state.current == s_in_escape:
115  if c == '\\':
116  state.escape_count += 1
117  elif c == '"':
118  nbs, escaped_delim = divmod(state.escape_count, 2)
119  state.argument.append(nbs * '\\')
120  if escaped_delim:
121  state.argument.append('"')
122  state.current = state.previous
123  else:
124  if state.previous == s_in_quotes:
125  state.current = s_free
126  else:
127  state.current = s_in_quotes
128  state.previous = s_in_escape
129  else:
130  state.argument.append(state.escape_count * '\\')
131  state.argument.append(c)
132  state.current = state.previous
133  state.previous = s_in_escape
134  if state.current != s_free:
135  raise ValueError('Illegal command line.')
136  return result

323 # This function runs the given command; waits for it to finish; then
324 # returns all output as a string. STDERR is included in output. If the full
325 # path to the command is not given then the path is searched.
326 #
327 # Note that lines are terminated by CR/LF (\\r\\n) combination even on
328 # UNIX-like systems because this is the standard for pseudo ttys. If you set
329 # 'withexitstatus' to true, then run will return a tuple of (command_output,
330 # exitstatus). If 'withexitstatus' is false then this returns just
331 # command_output.
332 #
333 # The run() function can often be used instead of creating a spawn instance.
334 # For example, the following code uses spawn::
335 #
336 # from pexpect import *
337 # child = spawn('scp foo myname@host.example.com:.')
338 # child.expect ('(?i)password')
339 # child.sendline (mypassword)
340 #
341 # The previous code can be replace with the following::
342 #
343 # from pexpect import *
344 # run ('scp foo myname@host.example.com:.', events={'(?i)password': mypassword})
345 #
346 # Examples
347 # ========
348 #
349 # Start the apache daemon on the local machine::
350 #
351 # from pexpect import *
352 # run ("/usr/local/apache/bin/apachectl start")
353 #
354 # Check in a file using SVN::
355 #
356 # from pexpect import *
357 # run ("svn ci -m 'automatic commit' my_file.py")
358 #
359 # Run a command and capture exit status::
360 #
361 # from pexpect import *
362 # (command_output, exitstatus) = run ('ls -l /bin', withexitstatus=1)
363 #
364 # Tricky Examples
365 # ===============
366 #
367 # The following will run SSH and execute 'ls -l' on the remote machine. The
368 # password 'secret' will be sent if the '(?i)password' pattern is ever seen::
369 #
370 # run ("ssh username@machine.example.com 'ls -l'", events={'(?i)password':'secret\\n'})
371 #
372 # This will start mencoder to rip a video from DVD. This will also display
373 # progress ticks every 5 seconds as it runs. For example::
374 #
375 # from pexpect import *
376 # def print_ticks(d):
377 # print d['event_count'],
378 # run ("mencoder dvd://1 -o video.avi -oac copy -ovc copy", events={TIMEOUT:print_ticks}, timeout=5)
379 #
380 # The 'events' argument should be a dictionary of patterns and responses.
381 # Whenever one of the patterns is seen in the command out run() will send the
382 # associated response string. Note that you should put newlines in your
383 # string if Enter is necessary. The responses may also contain callback
384 # functions. Any callback is function that takes a dictionary as an argument.
385 # The dictionary contains all the locals from the run() function, so you can
386 # access the child spawn object or any other variable defined in run()
387 # (event_count, child, and extra_args are the most useful). A callback may
388 # return True to stop the current run process otherwise run() continues until
389 # the next event. A callback may also return a string which will be sent to
390 # the child. 'extra_args' is not used by directly run(). It provides a way to
391 # pass data to a callback function through run() through the locals
392 # dictionary passed to a callback.
393 def run (command, timeout=-1, withexitstatus=False, events=None, extra_args=None, logfile=None, cwd=None, env=None, stub=None):
394 
395 
396  if timeout == -1:
397  child = winspawn(command, maxread=2000, logfile=logfile, cwd=cwd, env=env, stub=None)
398  else:
399  child = winspawn(command, timeout=timeout, maxread=2000, logfile=logfile, cwd=cwd, env=env, stub=None)
400  if events is not None:
401  patterns = list(events.keys())
402  responses = list(events.values())
403  else:
404  patterns=None # We assume that EOF or TIMEOUT will save us.
405  responses=None
406  child_result_list = []
407  event_count = 0
408  while 1:
409  try:
410  index = child.expect (patterns)
411  if isinstance(child.after, str):
412  child_result_list.append(child.before + child.after)
413  else: # child.after may have been a TIMEOUT or EOF, so don't cat those.
414  child_result_list.append(child.before)
415 
416  if isinstance(responses[index], str):
417  child.send(responses[index])
418  elif isinstance(responses[index], types.FunctionType):
419  callback_result = responses[index](locals())
420  #sys.stdout.flush()
421  if isinstance(callback_result, str):
422  child.send(callback_result)
423  elif callback_result:
424  child.expect(EOF)
425  break
426  else:
427  child.terminate()
428  raise TypeError ('The callback must be a string or function type.')
429  event_count = event_count + 1
430  except TIMEOUT as e:
431  child_result_list.append(child.before)
432  child.terminate()
433  break
434  except EOF as e:
435  child_result_list.append(child.before)
436  child.close()
437  break
438  child_result = ''.join(child_result_list)
439  if withexitstatus:
440  child.wait()
441  return (child_result, child.exitstatus)
442  else:
443  return child_result

447 # A version of pexpect.spawn for the Windows platform.
448 class winspawn(spawn):
449 
450  # The Windows version of spawn is quite different when compared to the
451  # Posix version.
452  #
453  # The first difference is that it's not possible on Windows to select()
454  # on a file descriptor that corresponds to a file or a pipe. Therefore,
455  # to do non-blocking I/O, we need to use threads.
456  #
457  # Secondly, there is no way to pass /only/ the file descriptors
458  # corresponding to the redirected stdin/out/err to the child. Either all
459  # inheritable file descriptors are passed, or none. We solve this by
460  # indirectly executing our child via a stub for which we close all file
461  # descriptors. The stub communicates back to us via a named pipe.
462  #
463  # Finally, Windows does not have ptys. It does have the concept of a
464  # "Console" though but it's much less sophisticated. This code runs the
465  # child in a new console by passing the flag CREATE_NEW_CONSOLE to
466  # CreateProcess(). We create a new console for our child because this
467  # way it cannot interfere with the current console, and it is also
468  # possible to run the main program without a console (e.g. a Windows
469  # service).
470  #
471  # NOTE:
472  # Some special application will identify the input type. If its input handle
473  # is not the stdin, the child process will disable the interactive mode.
474  # For example: To run python as interactive mode, we should do like below:
475  # child = winspawn('python', ['-i'])
476  # option '-i' will force the python into interactive mode
477  #
