2 # Pexpect is a Python module for spawning child applications and controlling
3 # them automatically. Pexpect can be used for automating interactive applications
4 # such as ssh, ftp, passwd, telnet, etc. It can be used to a automate setup
5 # scripts for duplicating software package installations on different servers. It
6 # can be used for automated software testing. Pexpect is in the spirit of Don
7 # Libes' Expect, but Pexpect is pure Python. Other Expect-like modules for Python
8 # require TCL and Expect or require C extensions to be compiled. Pexpect does not
9 # use C, Expect, or TCL extensions. It should work on any platform that supports
10 # the standard Python pty module. The Pexpect interface focuses on ease of use so
11 # that simple tasks are easy.
12 #
13 # There are two main interfaces to the Pexpect system; these are the function,
14 # run() and the class, spawn. The spawn class is more powerful. The run()
15 # function is simpler than spawn, and is good for quickly calling program. When
16 # you call the run() function it executes a given program and then returns the
17 # output. This is a handy replacement for os.system().
18 #
19 # For example::
20 #
21 # pexpect.run('ls -la')
22 #
23 # The spawn class is the more powerful interface to the Pexpect system. You can
24 # use this to spawn a child program then interact with it by sending input and
25 # expecting responses (waiting for patterns in the child's output).
26 #
27 # For example::
28 #
29 # child = pexpect.spawn('scp foo user@example.com:.')
30 # child.expect('Password:')
31 # child.sendline(mypassword)
32 #
33 # This works even for commands that ask for passwords or other input outside of
34 # the normal stdio streams. For example, ssh reads input directly from the TTY
35 # device which bypasses stdin.

82 # This function runs the given command; waits for it to finish; then
83 # returns all output as a string. STDERR is included in output. If the full
84 # path to the command is not given then the path is searched.
85 #
86 # Note that lines are terminated by CR/LF (\\r\\n) combination even on
87 # UNIX-like systems because this is the standard for pseudottys. If you set
88 # 'withexitstatus' to true, then run will return a tuple of (command_output,
89 # exitstatus). If 'withexitstatus' is false then this returns just
90 # command_output.
91 #
92 # The run() function can often be used instead of creating a spawn instance.
93 # For example, the following code uses spawn::
94 #
95 # from pexpect import *
96 # child = spawn('scp foo user@example.com:.')
97 # child.expect('(?i)password')
98 # child.sendline(mypassword)
99 #
100 # The previous code can be replace with the following::
101 #
102 # from pexpect import *
103 # run('scp foo user@example.com:.', events={'(?i)password': mypassword})
104 #
105 # **Examples**
106 #
107 # Start the apache daemon on the local machine::
108 #
109 # from pexpect import *
110 # run("/usr/local/apache/bin/apachectl start")
111 #
112 # Check in a file using SVN::
113 #
114 # from pexpect import *
115 # run("svn ci -m 'automatic commit' my_file.py")
116 #
117 # Run a command and capture exit status::
118 #
119 # from pexpect import *
120 # (command_output, exitstatus) = run('ls -l /bin', withexitstatus=1)
121 #
122 # The following will run SSH and execute 'ls -l' on the remote machine. The
123 # password 'secret' will be sent if the '(?i)password' pattern is ever seen::
124 #
125 # run("ssh username@machine.example.com 'ls -l'",
126 # events={'(?i)password':'secret\\n'})
127 #
128 # This will start mencoder to rip a video from DVD. This will also display
129 # progress ticks every 5 seconds as it runs. For example::
130 #
131 # from pexpect import *
132 # def print_ticks(d):
133 # print d['event_count'],
134 # run("mencoder dvd://1 -o video.avi -oac copy -ovc copy",
135 # events={TIMEOUT:print_ticks}, timeout=5)
136 #
137 # The 'events' argument should be either a dictionary or a tuple list that
138 # contains patterns and responses. Whenever one of the patterns is seen
139 # in the command output, run() will send the associated response string.
140 # So, run() in the above example can be also written as:
141 #
142 # run("mencoder dvd://1 -o video.avi -oac copy -ovc copy",
143 # events=[(TIMEOUT,print_ticks)], timeout=5)
144 #
145 # Use a tuple list for events if the command output requires a delicate
146 # control over what pattern should be matched, since the tuple list is passed
147 # to pexpect() as its pattern list, with the order of patterns preserved.
148 #
149 # Note that you should put newlines in your string if Enter is necessary.
150 #
151 # Like the example above, the responses may also contain a callback, either
152 # a function or method. It should accept a dictionary value as an argument.
153 # The dictionary contains all the locals from the run() function, so you can
154 # access the child spawn object or any other variable defined in run()
155 # (event_count, child, and extra_args are the most useful). A callback may
156 # return True to stop the current run process. Otherwise run() continues
157 # until the next event. A callback may also return a string which will be
158 # sent to the child. 'extra_args' is not used by directly run(). It provides
159 # a way to pass data to a callback function through run() through the locals
160 # dictionary passed to a callback.
161 #

181 def _run(command, timeout, withexitstatus, events, extra_args, logfile, cwd,
182  env, _spawn, **kwargs):
183  if timeout == -1:
184  child = _spawn(command, maxread=2000, logfile=logfile, cwd=cwd, env=env,
185  **kwargs)
186  else:
187  child = _spawn(command, timeout=timeout, maxread=2000, logfile=logfile,
188  cwd=cwd, env=env, **kwargs)
189  if isinstance(events, list):
190  patterns= [x for x,y in events]
191  responses = [y for x,y in events]
192  elif isinstance(events, dict):
193  patterns = list(events.keys())
194  responses = list(events.values())
195  else:
196  # This assumes EOF or TIMEOUT will eventually cause run to terminate.
197  patterns = None
198  responses = None
199  child_result_list = []
200  event_count = 0
201  while True:
202  try:
203  index = child.expect(patterns)
204  if isinstance(child.after, child.allowed_string_types):
205  child_result_list.append(child.before + child.after)
206  else:
207  # child.after may have been a TIMEOUT or EOF,
208  # which we don't want appended to the list.
209  child_result_list.append(child.before)
210  if isinstance(responses[index], child.allowed_string_types):
211  child.send(responses[index])
212  elif (isinstance(responses[index], types.FunctionType) or
213  isinstance(responses[index], types.MethodType)):
214  callback_result = responses[index](locals())
215  sys.stdout.flush()
216  if isinstance(callback_result, child.allowed_string_types):
217  child.send(callback_result)
218  elif callback_result:
219  break
220  else:
221  raise TypeError("parameter `event' at index {index} must be "
222  "a string, method, or function: {value!r}"
223  .format(index=index, value=responses[index]))
224  event_count = event_count + 1
225  except TIMEOUT:
226  child_result_list.append(child.before)
227  break
228  except EOF:
229  child_result_list.append(child.before)
230  break
231  child_result = child.string_type().join(child_result_list)
232  if withexitstatus:
233  child.close()
234  return (child_result, child.exitstatus)
235  else:
236  return child_result
